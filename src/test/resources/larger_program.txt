// function to find the biggest value out of three provided
fun biggestOfThreeNumbers(a, b, c) {
    if (a > b && a > c) {
        return a
    }

    if (b > a && b > c) {
        return b
    } else {
        return c
    }
}

fun iterate(a, b, c) {
    while (biggestOfThreeNumbers(a, b, c) != c) {
        c = c + 1
    }

    return c
}
